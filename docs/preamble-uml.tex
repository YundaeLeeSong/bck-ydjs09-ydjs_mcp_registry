\newcommand{\umldevicetitle}[1]{%
  \begin{tabular}{@{}c@{}}%
    \texttt{<<device>>}\\%
    :#1%
  \end{tabular}%
}

% #1 = SW, #2 = NE, #3 = device name
\newcommand{\umldeviceframe}[3]{%
  \draw[uml/device frame] (#1) rectangle (#2);
  \coordinate (uml@frame-nw) at (#1 |- #2);
  \node[uml/device title] at ($(uml@frame-nw)!0.5!(#2)$) [yshift=-9pt] {\umldevicetitle{#3}};
}









% =============================================================================
% Component diagrams  (\umlbasiccomponent + interface overrides below)

\tikzset{%
  uml/comp if line/.style={%
    line width=0.9pt, draw=\umlthemeDraw,
  },%
  uml/comp if label/.style={%
    font=\small, anchor=south, text=\umlthemeText, inner sep=1pt,
  },%
}

\newcommand{\umlCompIfStem}{1.6cm}    % listener: box -> O
\newcommand{\umlCompIfReqStem}{6.0cm} % caller:   box -> (
\newcommand{\umlCompIfBall}{0.18cm}
\newcommand{\umlCompIfSocket}{0.18cm}
\newcommand{\umlCompIfLabelY}{0.55cm}

\renewcommand{\umlprovidedinterface}[2][]{%
  \def\uml@interfacelabel{}%
  \IfSubStr{#1}{interface=}{%
    \StrBehind{#1}{interface=}[\uml@interfacelabel]%
  }{}%
  \coordinate (#2-prov) at ($(#2.west)+(-\umlCompIfStem,0)$);
  \draw[uml/comp if line] (#2.west) -- ($(#2-prov)!{\umlCompIfBall}!(#2.west)$);
  \filldraw[draw=\umlthemeDraw, fill=\umlthemeFill, line width=0.9pt] (#2-prov) circle (\umlCompIfBall);
  \node[uml/comp if label] at ($(#2-prov)+(0,\umlCompIfLabelY)$) {\uml@interfacelabel};
}

\renewcommand{\umlrequiredinterface}[2]{%
  \coordinate (#2-sock) at ($(#1.east)!{\umlCompIfReqStem}!(#2-prov)$);
  \draw[uml/comp if line] (#1.east) -- ($(#2-sock)!{\umlCompIfSocket}!(#1.east)$);
  \draw[uml/comp if line] (#2-sock) ++(0,\umlCompIfSocket) arc (90:270:\umlCompIfSocket);
}
